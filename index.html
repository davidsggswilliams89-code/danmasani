let testimonials = []; let tIndex = 0; let tAuto; /* LOAD FROM FIREBASE */ async function loadTestimonials(){ const snap = await getDocs(collection(db,"testimonials")); testimonials = []; snap.forEach(d=>{ testimonials.push({ id: d.id, image: d.data().image }); }); renderTestimonials(); } /* RENDER */ function renderTestimonials(){ const box = document.getElementById("testimonialCarousel"); const dots = document.getElementById("testimonialDots"); if(!box || !dots) return; box.innerHTML = ""; dots.innerHTML = ""; if(testimonials.length === 0){ box.innerHTML = `

No testimonials yet

`; return; } testimonials.forEach((t,i)=>{ box.innerHTML += ``; let d = document.createElement("span"); d.className = "dot" + (i===0?" active":""); d.onclick = ()=>scrollTestimonial(i); dots.appendChild(d); }); scrollTestimonial(0); startAuto(); } /* SCROLL */ function scrollTestimonial(i){ const box = document.getElementById("testimonialCarousel"); const dots = document.querySelectorAll("#testimonialDots .dot"); box.scrollTo({ left: box.clientWidth * i, behavior: "smooth" }); dots.forEach(d=>d.classList.remove("active")); if(dots[i]) dots[i].classList.add("active"); tIndex = i; } /* AUTO SLIDE */ function startAuto(){ clearInterval(tAuto); if(testimonials.length <= 1) return; tAuto = setInterval(()=>{ tIndex = (tIndex + 1) % testimonials.length; scrollTestimonial(tIndex); },4000); }
